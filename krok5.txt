Cała assercja:

     check(res, {
    'status is 200':               (r) => r.status === 200,
    'response contains pizza':     (r) => r.json().pizza !== undefined,
    'pizza has a name':            (r) => r.json().pizza.name.length > 0,
    'number of ingredients >= 2':  (r) => r.json().pizza.ingredients.length >= 2,
  });

  if (res.status === 200) {
    const pizza = res.json().pizza;
    pizzaCounter.add(1);
    ingredientTrend.add(pizza.ingredients.length);
    const isVegetarian = pizza.ingredients.every(i => i.vegetarian === true);
    vegetarianRate.add(isVegetarian);
  }
